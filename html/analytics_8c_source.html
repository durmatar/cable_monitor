87/** ***************************************************************************
88 * @brief Aproximate Angle with three length inputs
89 * @param [in] distance left
90 * @param [in] distance right
91 * @param [in] distance midle
92 * @return Angle between -45° and 45°
93 *****************************************************************************/
94float CALC_Angle(float left, float right, float middle){
95 left = left / middle;
96 right = right / middle;
97 middle = 1;
98 float x;
99 if (left < (middle-0.2)) {
100 x = -30;
101 //-45° to 0°
102 } else if (right < (middle-0.2)){
103 //0° - 45°
104 x = 30;
105 } else {
106 x = 0;
107 }
108 return x;
109}

112/** ***************************************************************************
113 * @brief Calculate electrical current from magnetic field
114 * @param [in] distance[m]
115 * @param [in] hall amplitude
116 * @return calculated current[A]
117 *****************************************************************************/
118float CALC_ElCurrent(float amplitude, float distance){
119 float I,B;
120 // Calculate electro-magnetic field strength
121 B = (((amplitude*CALC_ADCVOLTRESOLUTION)/CALC_AMPOPAMP)/CALC_AMPHALLSENS);
122
123 // Calculate current
124 I = CALC_PIDANDPERM*(distance/B);
125
126 //return current
127 return I;
128}

131/** ***************************************************************************
132 * @brief Convert amplitude strength to distance
133 * @param [in] pointer to distance LUT
134 * @param [in] pointer to amplitude strength LUT
135 * @param [in] measurement
136 * @return calculated distance
137 *
138 * Look up distance values corresponding to strength, if value is between two
139 * entries calculate exact distance with a linear function.
140 *****************************************************************************/
141float CALC_Distance(float* lutDistance, float* lutStrenght, float measurement){
142 int t,d,s;
143 float distance = -1;
144 t=0;
145 s=-10;
146 d=-10;
147
148 // Catch to high and to low values
149 if(lutStrenght[CALC_LUTSIZE-1]>measurement){
150 measurement = lutStrenght[CALC_LUTSIZE-1];
151 } else if (measurement>lutStrenght[0]) {
152 measurement = lutStrenght[0];
153 }
154
155 // Go through LUT
156 for(int i=0; i < CALC_LUTSIZE ; i++ ){
157 if (measurement == lutStrenght[i]){
158 t = i;
159 s=t;
160 }
161 if ( (measurement < lutStrenght[i])&& (measurement > lutStrenght[i+1])){
162 t = i;
163 d=t;
164 }
165 }
166
167 if(t==d){
168 float a = (lutDistance[t+1]-lutDistance[t])/(lutStrenght[t+1]-lutStrenght[t]);
169 distance = a*(measurement-lutStrenght[t]) + lutDistance[t];
170 }
171
172 if(t==s){
173 distance = lutDistance[t];
174 }
175
176 return distance;
177}

180/** ***************************************************************************
181 * @brief Calculate distance from measurement input and mode setting
182 * @param [in] measurement value
183 * @param [in] selected mode
184 * @param [in] channel selection, if true right side
185 * @return calculated distance

187float CALC_DistanceMode(float measurement, uint16_t mode, bool right){
188 float distance;
189 float* lut;
190 // Select left or right
191 if (right) {
192 lut = CALC_wpcRight[mode];
193 } else {
194 lut = CALC_wpcLeft[mode];
195 }
196 // Calculate distance with correct mode
197 distance = CALC_Distance(CALC_distanceLUT, lut, measurement);
198 return distance;
199}

203/** ***************************************************************************
204 * @brief Analytics handler
205 *
206 * Handler for the analytics processes. Start and stop measurements according
207 * to option inputs. Collect and analyse measurements. Output analysed data and
208 * trigger the required events.
209 *****************************************************************************/
210void ANA_Handler(void){
211 //start measurement with button input
212 if (ANA_inBtn & !ANA_measBusy) {
213 ANA_measBusy = true;
214 ANA_inBtn = false;
215 }
216
217 //collect data
218 if (ANA_wpcBusy & ANA_inMeasReady) {
219 ANA_wpcLeft[ANA_cycle]=(float)ANA_inAmpLeft;
220 ANA_wpcRight[ANA_cycle]=(float)ANA_inAmpRight;
221 } else if (ANA_hallBusy & ANA_inMeasReady){
222 ANA_hallLeft[ANA_cycle]=(float)ANA_inAmpLeft;
223 ANA_hallRight[ANA_cycle]=(float)ANA_inAmpRight;
224 }
225
226 //while meas busy, start measurements
227 if (ANA_measBusy){
228 //start wpc
229 if ((ANA_cycle < ANA_inOptn[3])&(!ANA_wpcBusy)&(!ANA_hallBusy)) {
230 ANA_outStartWPC = true;
231 ANA_wpcBusy = true;
232 //start hall
233 } else if (ANA_wpcBusy & ANA_inMeasReady){
234 ANA_outStartHALL = true;
235 ANA_wpcBusy = false;
236 ANA_hallBusy = true;
237 } else if (ANA_hallBusy & ANA_inMeasReady){
238 ANA_hallBusy = false;
239 ANA_cycle ++;
240 }
241 }
242
243 //When cycles finished or in continuous mode and single cycle is completed
244 if ((ANA_cycle == ANA_inOptn[3])&(!ANA_wpcBusy)&(!ANA_hallBusy)) {
245 //Analyse data
246 float mean,stdDeviation,angle,current;
247 mean = 0;
248 stdDeviation = 0;
249 angle = 0;
250 current = 0;
251 if (ANA_inOptn[1]==0) {
252 int accuracy = ANA_inOptn[3];
253 // Calculate distance
254 for (int i = 0; i < accuracy; ++i) {
255 ANA_wpcLeft[i]=CALC_DistanceMode(ANA_wpcLeft[i], ANA_inOptn[0], false);
256 ANA_wpcRight[i]=CALC_DistanceMode(ANA_wpcRight[i], ANA_inOptn[0], true);
257 }
258 // Sum
259 float sumLeft,sumRight;
260 for (int i = 0; i < accuracy; ++i) {
261 sumLeft = sumLeft+ANA_wpcLeft[i];
262 sumRight = sumRight+ANA_wpcRight[i];
263 }
264 // Mean
265 float meanLeft,meanRight;
266 meanLeft = sumLeft/accuracy;
267 meanRight = sumRight/accuracy;
268 mean = (sumLeft+sumRight)/(2*accuracy);
269
270 // Standard Deviation
271 if (accuracy>1){
272 // Deviations
273 float deviation[2*accuracy];
274 for (int i = 0; i < accuracy; ++i) {
275 // Left
276 deviation[i]=ANA_wpcLeft[i]-mean;
277 deviation[i]=deviation[i]*deviation[i];
278 // Right
279 deviation[i*2]=ANA_wpcRight[i]-mean;
280 deviation[i*2]=deviation[i*2]*deviation[i*2];
281 }
282 // Variation
283 float var = 0;
284 for (int i = 0; i < 2*accuracy; ++i) {
285 var = var + deviation[1];
286 }
287 var = var/(accuracy*2);
288 // Standard Deviation
289 stdDeviation = sqrtf(var);
290 }
291
292 // Angle
293 angle = CALC_Angle(meanRight, meanLeft, mean);
294
295 // Current
296 if ((mean<10)&(mean>0)) {
297 // Mean of hall sensor
298 float sumHall;
299 for (int i = 0; i < accuracy; ++i) {
300 sumHall = ANA_hallLeft[i]+sumHall;
301 sumHall = ANA_hallRight[i]+sumHall;
302 }
303 sumHall = sumHall/(2*accuracy);
304
305 current = CALC_ElCurrent(sumHall, (mean/1000));
306 }
307
308 // Transfer results
309 ANA_outResults[0]=angle; // Angle
310 ANA_outResults[1]=mean; // Distance
311 ANA_outResults[2]=stdDeviation; //Standard deviation
312 ANA_outResults[3]=current; //Current
313 ANA_outDataReady = true;
314
315 } else { //transfer raw data
316 // Calculate means of all 4 inputs
317 float meanWPCright, meanWPCleft, meanHALLright, meanHALLleft;
318 meanHALLleft = 0;
319 meanHALLright = 0;
320 meanWPCleft = 0;
321 meanWPCright = 0;
322
323 int accuracy = ANA_inOptn[3];
324
325 for (int i = 0; i < accuracy; ++i) {
326 meanHALLleft = meanHALLleft + ANA_hallLeft[i];
327 meanHALLright = meanHALLright + ANA_hallRight[i];
328 meanWPCleft = meanWPCleft + ANA_wpcLeft[i];
329 meanWPCright = meanWPCright + ANA_wpcRight[i];
330 }
331 meanHALLleft = meanHALLleft / accuracy;
332 meanHALLright = meanHALLright / accuracy;
333 meanWPCleft = meanWPCleft / accuracy;
334 meanWPCright = meanWPCright / accuracy;
335
336 // Transfer results
337 ANA_outResults[0]=meanHALLright; // HallRight
338 ANA_outResults[1]=meanHALLleft; // HallLeft
339 ANA_outResults[2]=meanWPCright; //WPCRight
340 ANA_outResults[3]=meanWPCleft; //WPCLeft
341 ANA_outDataReady = true;
342 }
343
344 }
345
346 //end measurement when cycles are finished
347 //or button is pressed when in streaming mode
348 if (((ANA_cycle == ANA_inOptn[3])&!(ANA_inOptn[2]==1))|(ANA_inBtn & (ANA_inOptn[2]==1))){
349 ANA_measBusy = false;
350 }
351
352 // Set cycles to zero if measurement finished
353 if (ANA_cycle == ANA_inOptn[3]) {
354 ANA_cycle = 0;
355 }
356
357 ANA_inBtn = false;
358 ANA_inMeasReady = false;
359}
