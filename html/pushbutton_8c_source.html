39/** ***************************************************************************
40 * @brief Configure the GPIO for the USER pushbutton
41 *
42 * The USER pushbutton is connected to PA0.
43 *****************************************************************************/
44void PB_init(void)
45{
46 __HAL_RCC_GPIOA_CLK_ENABLE(); // Enable Clock for GPIO port A
47 GPIOA->MODER |= (0u << GPIO_MODER_MODER0_Pos); // Pin 0 of port A = input
48}

51/** ***************************************************************************
52 * @brief Configure interrupt on rising edge for the USER pushbutton
53 *
54 * The USER pushbutton is connected to PA0.
55 *****************************************************************************/
56void PB_enableIRQ(void)
57{
58 __HAL_RCC_SYSCFG_CLK_ENABLE(); // Enable Clock for system config
59 SYSCFG->EXTICR[0] |= SYSCFG_EXTICR1_EXTI0_PA; // EXTI multiplexer
60 EXTI->RTSR |= EXTI_RTSR_TR0; // Rising Trigger Select on int. line 0
61 EXTI->IMR |= EXTI_IMR_MR0; // Interrupt Mask enable on int. line 0
62 NVIC_ClearPendingIRQ(EXTI0_IRQn); // Clear pending interrupt on line 0
63 NVIC_EnableIRQ(EXTI0_IRQn); // Enable interrupt line 0 in the NVIC
64}
