77/** ***************************************************************************
78 * @brief Configure GPIOs in analog mode.
79 *
80 * @note The input number for the ADCs is not equal to the GPIO pin number!
81 * - ADC3_IN4 = GPIO PF6 (wcp right)
82 * - ADC3_IN6 = GPIO PF8 (hall left)
83 * - ADC123_IN11 = GPIO PC1 (hall right)
84 * - ADC123_IN13 = GPIO PC3 (wpc left)
85 *****************************************************************************/
86void MEAS_GPIO_analog_init(void)
87{
88 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
89 GPIOF->MODER |= (GPIO_MODER_MODER6_Msk);// Analog mode for PF6 = ADC3_IN4
90 GPIOF->MODER |= (GPIO_MODER_MODER8_Msk);// Analog mode for PF8 = ADC3_IN6
91 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
92 GPIOC->MODER |= (GPIO_MODER_MODER1_Msk);// Analog mode for PC1 = ADC123_IN11
93 GPIOC->MODER |= (GPIO_MODER_MODER3_Msk);// Analog mode for PC3 = ADC123_IN13
94}

111 * @brief Configure the timer to trigger the ADC(s)
112 *
113 * @note For debugging purposes the timer interrupt might be useful.

115void MEAS_timer_init(void)
116{
117 __HAL_RCC_TIM2_CLK_ENABLE(); // Enable Clock for TIM2
118 TIM2->PSC = TIM_PRESCALE; // Prescaler for clock freq. = 1MHz
119 TIM2->ARR = TIM_TOP; // Auto reload = counter top value
120 TIM2->CR2 |= TIM_CR2_MMS_1; // TRGO on update
121 /* If timer interrupt is not needed, comment the following lines */
122 TIM2->DIER |= TIM_DIER_UIE; // Enable update interrupt
123 NVIC_ClearPendingIRQ(TIM2_IRQn); // Clear pending interrupt on line 0
124 NVIC_EnableIRQ(TIM2_IRQn); // Enable interrupt line 0 in the NVIC
125}

128/** ***************************************************************************
129 * @brief Initialize ADC, timer and DMA for WPC measurement
130 *
131 * Uses ADC3 and DMA2_Stream1 channel2
132 * @n The ADC3 trigger is set to TIM2 TRGO event
133 * @n At each trigger both inputs are converted sequentially
134 * and transfered to memory by the DMA.
135 * @n As each conversion triggers the DMA, the number of transfers is doubled.
136 * @n The DMA triggers the transfer complete interrupt when all data is ready.
137 * @n The inputs used are ADC123_IN13 = GPIO PC3 and ADC3_IN4 = GPIO PF6
138 *****************************************************************************/
139void ADC3_IN13_IN4_scan_init(void)
140{
141 __HAL_RCC_ADC3_CLK_ENABLE(); // Enable Clock for ADC3
142 ADC3->SQR1 |= ADC_SQR1_L_0; // Convert 2 inputs
143 ADC3->SQR3 |= (13UL << ADC_SQR3_SQ1_Pos); // Input 13 = first conversion
144 ADC3->SQR3 |= (4UL << ADC_SQR3_SQ2_Pos); // Input 4 = second conversion
145 ADC3->CR1 |= ADC_CR1_SCAN; // Enable scan mode
146 ADC3->CR2 |= (1UL << ADC_CR2_EXTEN_Pos); // En. ext. trigger on rising e.
147 ADC3->CR2 |= (6UL << ADC_CR2_EXTSEL_Pos); // Timer 2 TRGO event
148 ADC3->CR2 |= ADC_CR2_DMA; // Enable DMA mode
149 __HAL_RCC_DMA2_CLK_ENABLE(); // Enable Clock for DMA2
150 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA stream 1
151 while (DMA2_Stream1->CR & DMA_SxCR_EN) { ; } // Wait for DMA to finish
152 DMA2->LIFCR |= DMA_LIFCR_CTCIF1; // Clear transfer complete interrupt fl.
153 DMA2_Stream1->CR |= (2UL << DMA_SxCR_CHSEL_Pos); // Select channel 2
154 DMA2_Stream1->CR |= DMA_SxCR_PL_1; // Priority high
155 DMA2_Stream1->CR |= DMA_SxCR_MSIZE_1; // Memory data size = 32 bit
156 DMA2_Stream1->CR |= DMA_SxCR_PSIZE_1; // Peripheral data size = 32 bit
157 DMA2_Stream1->CR |= DMA_SxCR_MINC; // Increment memory address pointer
158 DMA2_Stream1->CR |= DMA_SxCR_TCIE; // Transfer complete interrupt enable
159 DMA2_Stream1->NDTR = 2*ADC_NUMS; // Number of data items to transfer
160 DMA2_Stream1->PAR = (uint32_t)&ADC3->DR; // Peripheral register address
161 DMA2_Stream1->M0AR = (uint32_t)ADC_samples; // Buffer memory loc. address
162
163}

169 * Uses ADC3 and DMA2_Stream1 channel2

173 * @n As each conversion triggers the DMA, the number of transfers is doubled.
174 * @n The DMA triggers the transfer complete interrupt when all data is ready.

177void ADC3_IN11_IN6_scan_init(void)
178{

230/** ***************************************************************************
231 * @brief Interrupt handler for the ADCs
232 *
233 * Reads one sample from the ADC3 DataRegister and transfers it to a buffer.
234 * @n Stops when ADC_NUMS samples have been read.
235 *****************************************************************************/
236void ADC_IRQHandler(void)
237{
238 if (ADC3->SR & ADC_SR_EOC) { // Check if ADC3 end of conversion
239 ADC_samples[ADC_sample_count++] = ADC3->DR; // Read input channel 1 only
240 if (ADC_sample_count >= ADC_NUMS) { // Buffer full
241 TIM2->CR1 &= ~TIM_CR1_CEN; // Disable timer
242 ADC3->CR2 &= ~ADC_CR2_ADON; // Disable ADC3
243 ADC_reset();
244 MEAS_data_ready = true;
245 }
246
247 }
248}

251/** ***************************************************************************
252 * @brief Interrupt handler for DMA2 Stream1
253 *
254 * The samples from the ADC3 have been transfered to memory by the DMA2 Stream1
255 * and are ready for processing.
256 *****************************************************************************/
257void DMA2_Stream1_IRQHandler(void)
258{
259 if (DMA2->LISR & DMA_LISR_TCIF1) { // Stream1 transfer compl. interrupt f.
260 NVIC_DisableIRQ(DMA2_Stream1_IRQn); // Disable DMA interrupt in the NVIC
261 NVIC_ClearPendingIRQ(DMA2_Stream1_IRQn);// Clear pending DMA interrupt
262 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA
263 while (DMA2_Stream1->CR & DMA_SxCR_EN) { ; } // Wait for DMA to finish
264 DMA2->LIFCR |= DMA_LIFCR_CTCIF1;// Clear transfer complete interrupt fl.
265 TIM2->CR1 &= ~TIM_CR1_CEN; // Disable timer
266 ADC3->CR2 &= ~ADC_CR2_ADON; // Disable ADC3
267 ADC3->CR2 &= ~ADC_CR2_DMA; // Disable DMA mode
268 ADC_reset();
269 MEAS_analyse_data();
270 MEAS_data_ready = true;
271 }

274/** ***************************************************************************
275 * @brief Analyse data to detect amplitude strength
276 *
277 * Get 10 highest and lowest measurements, convert lowest values to highest,
278 * calculate a mean and save the results.
279 *****************************************************************************/
280void MEAS_analyse_data(void)
281{
282 uint32_t buffer_left_channel[ADC_NUMS];
283 uint32_t buffer_right_channel[ADC_NUMS];
284 for (int i = 0; i < ADC_NUMS; ++i) {
285 buffer_left_channel[i] = ADC_samples[2*i];
286 buffer_right_channel[i] = ADC_samples[((2*i)+1)];
287 }
288
289 //sort arrays from low to high
290 uint32_t temp_left;
291 uint32_t temp_right;
292 for (int i = 0; i < ADC_NUMS; ++i) {
293 for (int j = i+1; j < ADC_NUMS; ++j) {
294 if (buffer_left_channel[i]>buffer_left_channel[j]) {
295 temp_left = buffer_left_channel[i];
296 buffer_left_channel[i]=buffer_left_channel[j];
297 buffer_left_channel[j]=temp_left;
298 }
299 if (buffer_right_channel[i]>buffer_right_channel[j]) {
300 temp_right = buffer_right_channel[i];
301 buffer_right_channel[i]=buffer_right_channel[j];
302 buffer_right_channel[j]=temp_right;
303 }
304 }
305 }
306
307
308 uint32_t values_left[10];
309 uint32_t values_right[10];
310 //select 5 lowest values
311 for (int i = 0; i < 5; ++i) {
312 values_left[i]=buffer_left_channel[i];
313 values_right[i]=buffer_right_channel[i];
314 }
315
316 //select 5 highest values
317 for (int i = 0; i < 5; ++i) {
318 values_left[i+5]=buffer_left_channel[i+55];
319 values_right[i+5]=buffer_right_channel[i+55];
320 }
321
322 //check if values mean around middle of ADC range
323 uint32_t check_sum_left = 0;
324 uint32_t check_sum_right = 0;
325 for (int i = 0; i < 10; ++i) {
326 check_sum_left += values_left[i];
327 check_sum_right += values_right[i];
328 }
329 check_sum_left = check_sum_left / 10;
330 check_sum_right = check_sum_right / 10;
331
332 //convert low values to high values
333 for (int i = 0; i < 5; ++i) {
334 values_left[i] = ADC_MAX_VALUE - values_left[i];
335 values_right[i] = ADC_MAX_VALUE - values_right[i];
336 }
337
338 //calculate mean of all 10 values
339 uint32_t sum_left = 0;
340 uint32_t sum_right = 0;
341 for (int i = 0; i < 10; ++i) {
342 sum_left += values_left[i];
343 sum_right += values_right[i];
344 }
345 sum_left = sum_left / 10;
346 sum_right = sum_right / 10;
347
348 MEAS_amplitude_left = sum_left-(ADC_MAX_VALUE/2);
349 MEAS_amplitude_right = sum_right-(ADC_MAX_VALUE/2);
350}
