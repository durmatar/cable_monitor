52int main(void) {

59 BSP_LCD_SelectLayer(LCD_FOREGROUND_LAYER);
60 BSP_LCD_DisplayOn();
61 BSP_LCD_Clear(LCD_COLOR_WHITE);

63 BSP_TS_Init(BSP_LCD_GetXSize(), BSP_LCD_GetYSize()); // Touchscreen
64 /* Uncomment next line to enable touchscreen interrupt */
65 // BSP_TS_ITConfig(); // Enable Touchscreen interrupt
66
67 PB_init(); // Initialize the user pushbutton
68 PB_enableIRQ(); // Enable interrupt on user pushbutton
69
70 BSP_LED_Init(LED3); // Toggles in while loop
71 BSP_LED_Init(LED4); // Is toggled by user button
72
73 MENU_draw(); // Draw the menu
74 MENU_hint(); // Show hint at startup
75
76 gyro_disable(); // Disable gyro, use those analog inputs
77
78 MEAS_GPIO_analog_init(); // Configure GPIOs in analog mode
79 MEAS_timer_init(); // Configure the timer
80
81 /* Infinite while loop */
82 while (1) { // Infinitely loop in main function
83 BSP_LED_Toggle(LED3); // Visual feedback when running
84
85 if (MEAS_data_ready) { // Show data if new data available
86 MEAS_data_ready = false;
87 MEAS_show_data();
88 }
89
90 if (PB_pressed()) { // Check if user pushbutton was pressed
91 BSP_LED_Toggle(LED4);
92 }
93
94 /* Comment next line if touchscreen interrupt is enabled */
95 MENU_check_transition();
96
97 switch (MENU_get_transition()) { // Handle user menu choice
98 case MENU_NONE: // No transition => do nothing
99 break;
100 case MENU_ZERO: //HALL PC1 PF8
101 ADC3_IN11_IN6_scan_init();
102 ADC3_dual_scan_start();
103 break;
104 case MENU_ONE: //WPC PC3 PF6
105 ADC3_IN13_IN4_scan_init();
106 ADC3_dual_scan_start();
107 break;
108 default: // Should never occur
109 break;
110 }
111
112 HAL_Delay(50); // Wait or sleep
113 }

133 RCC_OscInitStruct.PLL.PLLM = 8;

157/** ***************************************************************************
158 * @brief Disable the GYRO on the microcontroller board.
159 *
160 * @note MISO of the GYRO is connected to PF8 and CS to PC1.
161 * @n Some times the GYRO goes into an undefined mode at startup
162 * and pulls the MISO low or high thus blocking the analog input on PF8.
163 * @n The simplest solution is to pull the CS of the GYRO low for a short while
164 * which is done with the code below.
165 * @n PF8 is also reconfigured.
166 * @n An other solution would be to remove the GYRO
167 * from the microcontroller board by unsoldering it.
168 *****************************************************************************/
169static void gyro_disable(void)
170{
171 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
172 /* Disable PC1 and PF8 first */
173 GPIOC->MODER &= ~GPIO_MODER_MODER1; // Reset mode for PC1
174 GPIOC->MODER |= GPIO_MODER_MODER1_0; // Set PC1 as output
175 GPIOC->BSRR |= GPIO_BSRR_BR1; // Set GYRO (CS) to 0 for a short time
176 HAL_Delay(10); // Wait some time
177 GPIOC->MODER |= GPIO_MODER_MODER1_Msk; // Analog mode PC1 = ADC123_IN11
178 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
179 GPIOF->OSPEEDR &= ~GPIO_OSPEEDR_OSPEED8; // Reset speed of PF8
180 GPIOF->AFR[1] &= ~GPIO_AFRH_AFSEL8; // Reset alternate func. of PF8
181 GPIOF->PUPDR &= ~GPIO_PUPDR_PUPD8; // Reset pulup/down of PF8
182 HAL_Delay(10); // Wait some time
183 GPIOF->MODER |= GPIO_MODER_MODER8_Msk; // Analog mode for PF6 = ADC3_IN4
184}
