102 ADC3_IN13_IN4_scan_init();
103 ADC3_dual_scan_start();
104 ANA_outStartWPC = false; // Reset wpc start event
105 }
106
107 if (ANA_outDataReady) { // Analytics data ready
108 // Transfer Data
109 if (ANA_inOptn[1]==0) {
110 // Analysed
111 if (ANA_outResults[1]<300) {
112 // Data usable
113 GUI_angle = ANA_outResults[0];
114 GUI_distance = ANA_outResults[1];
115 GUI_distanceDeviation = ANA_outResults[2];
116 GUI_current = ANA_outResults[3];

120 GUI_angle = 100;
121 GUI_distance = -1;
122 GUI_distanceDeviation = -1;
123 GUI_current = -1;
124 GUI_cable_not_detected = true;
125 }
126
127
128 } else {
129 // Raw
130 GUI_rawHallRight = ANA_outResults[0];
131 GUI_rawHallLeft = ANA_outResults[1];
132 GUI_rawWpcRight = ANA_outResults[2];
133 GUI_rawWpcLeft = ANA_outResults[3];
134 }
135 GUI_inputMeasReady = true;
136 ANA_outDataReady = false;
137 }
138
139 // Show measurement state on led
140 if (ANA_measBusy) {
141 BSP_LED_On(LED4);
142 } else {
143 BSP_LED_Off(LED4);
144 }
145
146 if (GUI_outOptn) { // Check if Options were changed
147 ANA_inOptn[0]=GUI_mode; // Transfer mode
148 ANA_inOptn[1]=GUI_options[0].active;// Transfer data type
149 ANA_inOptn[2]=GUI_options[1].active;// Transfer measuring type
150 switch (GUI_options[2].active) { // Transfer accuracy
151 case 0:
152 ANA_inOptn[3]=1;
153 break;
154 case 1:
155 ANA_inOptn[3]=5;
156 break;
157 case 2:
158 ANA_inOptn[3]=10;
159 break;
160 default:
161 break;
162 }
163 GUI_outOptn = false; // Reset option bit
164 }
165
166 //Analytics handler
167 ANA_Handler();
168
169 //Site handler
170 GUI_SiteHandler();
171 }

179static void SystemClock_Config(void){
180 RCC_OscInitTypeDef RCC_OscInitStruct = {0};
181 RCC_ClkInitTypeDef RCC_ClkInitStruct = {0};
182 RCC_PeriphCLKInitTypeDef PeriphClkInitStruct = {0};
183 /* Configure the main internal regulator output voltage */
184 __HAL_RCC_PWR_CLK_ENABLE();
185 __HAL_PWR_VOLTAGESCALING_CONFIG(PWR_REGULATOR_VOLTAGE_SCALE1);
186 /* Initialize High Speed External Oscillator and PLL circuits */
187 RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;
188 RCC_OscInitStruct.HSEState = RCC_HSE_ON;
189 RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
190 RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
191 RCC_OscInitStruct.PLL.PLLM = 8;
192 RCC_OscInitStruct.PLL.PLLN = 336;
193 RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV2;
194 RCC_OscInitStruct.PLL.PLLQ = 7;
195 HAL_RCC_OscConfig(&RCC_OscInitStruct);
196 /* Initialize gates and clock dividers for CPU, AHB and APB busses */
197 RCC_ClkInitStruct.ClockType = RCC_CLOCKTYPE_HCLK | RCC_CLOCKTYPE_SYSCLK
198 | RCC_CLOCKTYPE_PCLK1 | RCC_CLOCKTYPE_PCLK2;
199 RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
200 RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
201 RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV4;
202 RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV2;
203 HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_5);
204 /* Initialize PLL and clock divider for the LCD */
205 PeriphClkInitStruct.PeriphClockSelection = RCC_PERIPHCLK_LTDC;
206 PeriphClkInitStruct.PLLSAI.PLLSAIN = 192;
207 PeriphClkInitStruct.PLLSAI.PLLSAIR = 4;
208 PeriphClkInitStruct.PLLSAIDivR = RCC_PLLSAIDIVR_8;
209 HAL_RCCEx_PeriphCLKConfig(&PeriphClkInitStruct);
210 /* Set clock prescaler for ADCs */
211 ADC->CCR |= ADC_CCR_ADCPRE_0;
212}

215/** ***************************************************************************
216 * @brief Disable the GYRO on the microcontroller board.
217 *
218 * @note MISO of the GYRO is connected to PF8 and CS to PC1.
219 * @n Some times the GYRO goes into an undefined mode at startup
220 * and pulls the MISO low or high thus blocking the analog input on PF8.
221 * @n The simplest solution is to pull the CS of the GYRO low for a short while
222 * which is done with the code below.
223 * @n PF8 is also reconfigured.
224 * @n An other solution would be to remove the GYRO
225 * from the microcontroller board by unsoldering it.
226 *****************************************************************************/
227static void gyro_disable(void)
228{
229 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
230 /* Disable PC1 and PF8 first */
231 GPIOC->MODER &= ~GPIO_MODER_MODER1; // Reset mode for PC1
232 GPIOC->MODER |= GPIO_MODER_MODER1_0; // Set PC1 as output
233 GPIOC->BSRR |= GPIO_BSRR_BR1; // Set GYRO (CS) to 0 for a short time
234 HAL_Delay(10); // Wait some time
235 GPIOC->MODER |= GPIO_MODER_MODER1_Msk; // Analog mode PC1 = ADC123_IN11
236 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
237 GPIOF->OSPEEDR &= ~GPIO_OSPEEDR_OSPEED8; // Reset speed of PF8
238 GPIOF->AFR[1] &= ~GPIO_AFRH_AFSEL8; // Reset alternate func. of PF8
239 GPIOF->PUPDR &= ~GPIO_PUPDR_PUPD8; // Reset pulup/down of PF8
240 HAL_Delay(10); // Wait some time
241 GPIOF->MODER |= GPIO_MODER_MODER8_Msk; // Analog mode for PF6 = ADC3_IN4
242}
