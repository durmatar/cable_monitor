58/** ***************************************************************************
59 * @brief Draw the menu onto the display.
60 *
61 * Each menu entry has two lines.
62 * Text and background colors are applied.
63 * @n These attributes are defined in the variable MENU_draw[].
64 *****************************************************************************/
65void MENU_draw(void)
66{
67 BSP_LCD_SetFont(MENU_FONT);
68 uint32_t x, y, m, w, h;
69 y = MENU_Y;
70 m = MENU_MARGIN;
71 w = BSP_LCD_GetXSize()/MENU_ENTRY_COUNT;
72 h = MENU_HEIGHT;
73 for (uint32_t i = 0; i < MENU_ENTRY_COUNT; i++) {
74 x = i*w;
75 BSP_LCD_SetTextColor(MENU_entry[i].back_color);
76 BSP_LCD_FillRect(x+m, y+m, w-2*m, h-2*m);
77 BSP_LCD_SetBackColor(MENU_entry[i].back_color);
78 BSP_LCD_SetTextColor(MENU_entry[i].text_color);
79 BSP_LCD_DisplayStringAt(x+3*m, y+3*m,
80 (uint8_t *)MENU_entry[i].line1, LEFT_MODE);
81 BSP_LCD_DisplayStringAt(x+3*m, y+h/2,
82 (uint8_t *)MENU_entry[i].line2, LEFT_MODE);
83 }
84}

154/** ***************************************************************************
155 * @brief Check for selection/transition
156 *
157 * If the last transition has been consumed (MENU_NONE == MENU_transition)
158 * and the touchscreen has been touched for a defined period
159 * the variable MENU_transition is set to the touched item.
160 * @note Evalboard revision E (blue PCB) has an inverted y-axis
161 * in the touch controller compared to the display.
162 * Uncomment or comment the <b>\#define EVAL_REV_E</b> in main.h accordingly.
163 *****************************************************************************/
164void MENU_check_transition(void)
165{
166 static MENU_item_t item_old = MENU_NONE;
167 static MENU_item_t item_new = MENU_NONE;
168 static TS_StateTypeDef TS_State; // State of the touch controller
169 BSP_TS_GetState(&TS_State); // Get the state
170#ifdef EVAL_REV_E
171// Evalboard revision E (blue) has an inverted y-axis in the touch controller
172 TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
173#endif
174 if (TS_State.TouchDetected) { // If a touch was detected
175 /* Do only if last transition not pending anymore */
176 if (MENU_NONE == MENU_transition) {
177 item_old = item_new; // Store old item
178 /* If touched within the menu bar? */
179 if ((MENU_Y < TS_State.Y) && (MENU_Y+MENU_HEIGHT > TS_State.Y)) {
180 item_new = TS_State.X // Calculate new item
181 / (BSP_LCD_GetXSize()/MENU_ENTRY_COUNT);
182 if ((0 > item_new) || (MENU_ENTRY_COUNT <= item_new)) {
183 item_new = MENU_NONE; // Out of bounds
184 }
185 if (item_new == item_old) { // 2 times the same menu item
186 item_new = MENU_NONE;
187 MENU_transition = item_old;
188 }
189 }
190 }
191 }
192}

196/** ***************************************************************************
197 * @brief Interrupt handler for the touchscreen
198 *
199 * @note BSP_TS_ITConfig(); must be called in the main function
200 * to enable touchscreen interrupt.
201 * @note There are timing issues when interrupt is enabled.
202 * It seems that polling is the better choice with this evaluation board.
203 * @n Call MENU_check_transition() from the while loop in main for polling.
204 *
205 * The touchscreen interrupt is connected to PA15.
206 * @n The interrupt handler for external line 15 to 10 is called.
207 *****************************************************************************/
208void EXTI15_10_IRQHandler(void)
209{
210 if (EXTI->PR & EXTI_PR_PR15) { // Check if interrupt on touchscreen
211 EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
212 if (BSP_TS_ITGetStatus()) { // Get interrupt status
213 BSP_TS_ITClear(); // Clear touchscreen controller int.
214 MENU_check_transition();
215 }
216 EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
217 }
218}
